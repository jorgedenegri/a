endfunction function I=trapecios(f,a,b,n) h=(b-a)/n; I=0; for k=1:n I=I+f(a+(k-1)*h)+f(a+k*h); endfor I=I*h/2; endfunction function A = romberg(f,a,b,error) r(1,1) = trapecios(f,a,b,1); r(2,1) = trapecios(f,a,b,2);
